## 2. New Message Types for Chat Panel

### 2.1 ChatMessage Type Extension

```typescript
// Extend existing message types in aiAdvisorStore
type ChatMessageType =
  // Existing
  | 'user' | 'assistant' | 'error'
  // Proactive (existing)
  | 'login' | 'tip' | 'reminder' | 'recurring_confirm'
  | 'contract_expired' | 'health_check' | 'tax_form_review'
  | 'upload_success' | 'upload_review' | 'upload_error'
  | 'asset_confirm' | 'employer_month_confirm'
  // NEW: In-chat suggestions
  | 'suggestion'          // Rendered as structured card in chat
  // NEW: Follow-up questions
  | 'follow_up'           // Rendered with inline form controls
  // NEW: Processing progress
  | 'processing_update';  // Rendered with progress indicator

// === Idempotency Key ===
// All messages carry an idempotencyKey to prevent duplicate insertions
// during polling retries or concurrent uploads.
// Format: `${documentId}:${suggestionType}:${phase}`
// Example: "42:create_asset:suggestion" or "42:create_asset:processing_ocr"
// The store's pushMessage() deduplicates by this key — if a message with the
// same key already exists, it updates in-place instead of inserting a new one.
//
// ⚠️ HARD RULE: The idempotencyKey is GENERATED BY THE BACKEND and returned
// in the process-status response. The frontend MUST use the backend-provided key
// verbatim — it must NOT self-generate or infer keys by string concatenation.
// This prevents key drift when phase names or suggestion types change.

interface BaseChatMessage {
  id: string;                // UUID, auto-generated
  idempotencyKey: string;    // Stable key for deduplication
  timestamp: number;
}

interface SuggestionChatMessage extends BaseChatMessage {
  type: 'suggestion';
  suggestionType: string;   // 'create_property' | 'create_asset' | etc.
  documentId: string;
  extractedData: Record<string, any>;
  followUpQuestions?: FollowUpQuestion[];
  status: 'pending' | 'confirmed' | 'dismissed' | 'needs_input';
  // Unified action descriptor from backend (see Section 3.5)
  action?: ActionDescriptor;
}

interface FollowUpQuestion {
  id: string;
  question: string;          // "When did you start using this vehicle for business?"
  inputType: 'text' | 'number' | 'date' | 'select' | 'boolean';
  options?: { value: string; label: string }[];  // For select type
  defaultValue?: any;
  required: boolean;
  fieldKey: string;          // Maps to backend field: 'put_into_use_date'
  // Optional inline help — shown as subtle subtext beneath the input.
  // Saves the user from having to ask the AI or leave the flow.
  // Example: "The date you started using this item for your business, not the purchase date."
  // Example: "Standard split is 70% building / 30% land. Use your Liegenschaftsgutachten if available."
  helpText?: string;
  validation?: {
    min?: number;
    max?: number;
    pattern?: string;
  };
}

interface ProcessingUpdateMessage extends BaseChatMessage {
  type: 'processing_update';
  documentId: string;
  phase: 'uploading' | 'ocr' | 'classifying' | 'extracting' | 'analyzing' | 'complete';
  documentType?: string;     // Set after classification
  message: string;           // Human-readable status
  // Frontend-friendly unified state (see Section 3.1.1)
  uiState: UIState;
}

// === Unified UI State Enum ===
// Instead of the frontend deriving state from multiple fields (phase + suggestion.status +
// follow_up_questions.length), the backend returns a single stable enum.
type UIState =
  | 'processing'        // Document is being analyzed
  | 'needs_input'       // Follow-up questions need user answers
  | 'ready_to_confirm'  // All data collected, waiting for user confirm
  | 'confirmed'         // User confirmed, entity created
  | 'dismissed'         // User dismissed suggestion
  | 'error';            // Processing or confirmation failed

// === Unified Action Descriptor ===
// Proactive messages and suggestions carry this self-describing action contract,
// so the frontend doesn't need an ever-growing switch statement.
// Frozen enum of action kinds — new types must be explicitly added here.
// This prevents kind from drifting into free-form strings.
type ActionKind =
  | 'confirm_property'
  | 'confirm_asset'
  | 'confirm_recurring'
  | 'confirm_recurring_expense'
  | 'confirm_tax_data'
  | 'confirm_loan'
  | 'confirm_bank_transactions'
  | 'dismiss_suggestion';

interface ActionDescriptor {
  kind: ActionKind;      // Frozen enum — must be explicitly extended for new types
  targetId: string;      // document ID or entity ID
  endpoint: string;      // e.g., '/documents/42/confirm-asset'
  method: 'POST' | 'PUT' | 'DELETE';  // HTTP method
  payload?: Record<string, any>;       // Optional pre-filled data
  // Button labels — allows backend to control button text per action type
  // Falls back to generic i18n keys if not provided
  confirmLabel?: string;   // e.g., "Create Property" | "Import Tax Data" | "Confirm Recurring"
  dismissLabel?: string;   // e.g., "Not a Property" | "Skip" | "Not Recurring"
  detailLabel?: string;    // e.g., "View Details" | "Show Extracted Data"
}
```

**Rendering rules:**
- Simple suggestions (property, recurring, tax form): Show key data + Confirm/Dismiss buttons
- **Complex suggestions (asset, property with missing data): Show key data summary + Confirm/Dismiss buttons ONLY.** The detailed input collection is handled by a separate `follow_up` message, NOT embedded in the suggestion card. This prevents the chat from becoming "a form system inside a chat bubble".
- Bank statement: Show transaction list with checkboxes + Confirm Selected button

> **Product Rule**: ChatSuggestionCard is a **summary + action** component, not a form. All multi-field data collection goes through `ChatFollowUpQuestion` as a separate message. This keeps the chat conversational — the AI asks questions, the user answers, then the AI presents the final suggestion for confirmation.

```python
# NEW endpoint: GET /documents/{id}/process-status
# Returns current processing phase for real-time progress

@router.get("/{document_id}/process-status")
async def get_process_status(document_id: int, ...):
    """Return current processing phase for a document."""
    doc = db.query(Document).get(document_id)

    suggestion = _get_pending_suggestion(doc) if doc.pipeline_current_state == 'completed' else None

    # Derive a single ui_state for the frontend (see requirements NFR-6, FR-19)
    ui_state = _derive_ui_state(doc, suggestion)

    return {
        "phase": doc.pipeline_current_state,  # existing field
        "document_type": doc.document_type,
        "message": _phase_to_message(doc.pipeline_current_state, lang),
        "ui_state": ui_state,  # NEW: stable enum for frontend
        "suggestion": suggestion,
        # Phase timing — enables frontend to show "Processing for 12s...", detect stalls, trigger retries
        "phase_started_at": doc.pipeline_phase_started_at.isoformat() if doc.pipeline_phase_started_at else None,
        "phase_updated_at": doc.pipeline_phase_updated_at.isoformat() if doc.pipeline_phase_updated_at else None,
        # Retry tracking — if Phase 2 retries, frontend/logs know which attempt this is
        "current_phase_attempt": doc.pipeline_phase_attempt or 1,
        # Suggestion version — for concurrent write protection on follow-up answers
        "suggestion_version": suggestion.get("version") if suggestion else None,
        # Idempotency key — BACKEND is source of truth, frontend must NOT self-generate
        "idempotency_key": f"{document_id}:{suggestion['type'] if suggestion else 'none'}:{doc.pipeline_current_state}"
    }

def _derive_ui_state(doc, suggestion) -> str:
    """Single source of truth for frontend UI state."""
    if doc.pipeline_current_state in ('processing_phase_1', 'first_result_available', 'finalizing'):
        return 'processing'
    if doc.pipeline_current_state == 'phase_2_failed':
        return 'error'
    if not suggestion:
        return 'confirmed'  # No suggestion = already processed
    if suggestion.get('status') == 'confirmed':
        return 'confirmed'
    if suggestion.get('status') == 'dismissed':
        return 'dismissed'
    if suggestion.get('follow_up_questions') and len(suggestion['follow_up_questions']) > 0:
        return 'needs_input'
    return 'ready_to_confirm'
```

### 3.3 Follow-Up Answer Endpoint

```python
# NEW endpoint: POST /documents/{id}/follow-up
# Accepts user answers to follow-up questions and updates suggestion

@router.post("/{document_id}/follow-up")
async def submit_follow_up_answers(
    document_id: int,
    body: FollowUpAnswerRequest,
    # body.answers: Dict[str, Any]    # {"put_into_use_date": "2025-01-15", "business_use_percentage": 80}
    # body.use_defaults: bool = False  # If True, apply defaults for ALL unanswered questions
    # body.suggestion_version: int     # Optimistic concurrency check
    ...
):
    """
    Update suggestion with user-provided follow-up answers.

    Supports three modes:
    1. Full answers: All questions answered → suggestion becomes ready_to_confirm
    2. Partial answers: Some questions answered → remaining questions re-presented (order preserved)
    3. Use defaults: body.use_defaults=True → apply default_value for all unanswered questions

    Concurrent write protection:
    - Request must include suggestion_version matching the current version
    - If versions don't match (another tab/session modified it), return HTTP 409 Conflict
    """
    doc = db.query(Document).get(document_id)

    ocr_result = doc.ocr_result or {}
    suggestion = ocr_result.get("import_suggestion", {})

    # Optimistic concurrency check
    current_version = suggestion.get("version", 0)
    if body.suggestion_version is not None and body.suggestion_version != current_version:
        raise HTTPException(
            status_code=409,
            detail={
                "error": "suggestion_version_mismatch",
                "current_version": current_version,
                "message": "Suggestion was modified by another session. Please refresh and try again."
            }
        )
    suggestion_data = suggestion.get("data", {})
    all_questions = suggestion.get("follow_up_questions", [])

    # Apply explicit user answers
    suggestion_data.update(body.answers)

    # Track which defaults were applied (for chat feedback message)
    applied_defaults = {}

    if body.use_defaults:
        # Apply default_value for all questions NOT in body.answers
        for q in all_questions:
            if q["field_key"] not in body.answers and q.get("default_value") is not None:
                suggestion_data[q["field_key"]] = q["default_value"]
                applied_defaults[q["field_key"]] = {
                    "value": q["default_value"],
                    "question": q["question"],  # For the chat to display "I used standard 70/30 ratio"
                }
        remaining_questions = []
    else:
        # Only clear answered questions, preserve order of remaining
        answered_keys = set(body.answers.keys())
        remaining_questions = [
            q for q in all_questions
            if q["field_key"] not in answered_keys
        ]

    suggestion["follow_up_questions"] = remaining_questions
    suggestion["data"] = suggestion_data

    # Derive new status
    if not remaining_questions:
        suggestion["status"] = "ready_to_confirm"
    else:
        suggestion["status"] = "needs_input"

    # Bump version for optimistic concurrency
    suggestion["version"] = current_version + 1

    doc.ocr_result = ocr_result
    db.commit()

    return {
        "status": "updated",
        "ui_state": "ready_to_confirm" if not remaining_questions else "needs_input",
        "suggestion_version": suggestion["version"],  # Updated version for next request
        "remaining_questions": len(remaining_questions),
        "remaining_question_list": remaining_questions,  # So frontend can re-render
        "applied_defaults": applied_defaults,  # So chat can say "I used default X for Y"
    }
```

### 3.5 Unified Action Contract for Proactive Messages

```python
# All proactive messages and suggestions include a self-describing action descriptor.
# This prevents the frontend from needing an ever-growing switch statement.

def _build_action_descriptor(suggestion_type: str, document_id: int, entity_id: int = None) -> dict:
    """Build a unified action descriptor for the frontend."""
    ACTION_MAP = {
        "create_property": {
            "kind": "confirm_property",
            "endpoint": f"/documents/{document_id}/confirm-property",
            "method": "POST",
        },
        "create_asset": {
            "kind": "confirm_asset",
            "endpoint": f"/documents/{document_id}/confirm-asset",
            "method": "POST",
        },
        "create_recurring_income": {
            "kind": "confirm_recurring",
            "endpoint": f"/documents/{document_id}/confirm-recurring",
            "method": "POST",
        },
        "create_recurring_expense": {
            "kind": "confirm_recurring_expense",
            "endpoint": f"/documents/{document_id}/confirm-recurring-expense",
            "method": "POST",
        },
        "create_loan": {
            "kind": "confirm_loan",
            "endpoint": f"/documents/{document_id}/confirm-loan",
            "method": "POST",
        },
    }

    # All import_* types use the same endpoint
    if suggestion_type.startswith("import_"):
        return {
            "kind": "confirm_tax_data",
            "targetId": str(document_id),
            "endpoint": f"/documents/{document_id}/confirm-tax-data",
            "method": "POST",
        }

    base = ACTION_MAP.get(suggestion_type, {})
    return {
        **base,
        "targetId": str(entity_id or document_id),
    }
```

### 4.2 Document Upload → Chat Integration

```typescript
// In DocumentUpload.tsx or DocumentsPage.tsx, after successful upload:

async function handleUploadComplete(document: Document) {
  const store = useAIAdvisorStore.getState();

  // Step 1: Push processing message to chat (with idempotency key)
  store.pushMessage({
    type: 'processing_update',
    documentId: document.id,
    idempotencyKey: `${document.id}:none:processing_ocr`,  // Dedup key
    phase: 'ocr',
    message: t('ai.processing.analyzing'),  // "Analyzing your document..."
  });

  // Step 2: Poll for processing status
  const pollInterval = setInterval(async () => {
    const status = await documentService.getProcessStatus(document.id);

    // Update processing message
    store.updateProcessingMessage(document.id, {
      phase: status.phase,
      documentType: status.document_type,
      message: status.message,
    });

    // Step 3: When complete, show suggestion in chat
    if (status.phase === 'complete' && status.suggestion) {
      clearInterval(pollInterval);

      // Remove processing message
      store.removeProcessingMessage(document.id);

      // Push suggestion as chat message
      store.pushMessage({
        type: 'suggestion',
        suggestionType: status.suggestion.type,
        documentId: document.id,
        extractedData: status.suggestion.data,
        followUpQuestions: status.suggestion.follow_up_questions,
        status: status.suggestion.follow_up_questions?.length > 0
          ? 'needs_input'
          : 'pending',
      });

      // If follow-up questions exist, push them as separate message
      if (status.suggestion.follow_up_questions?.length > 0) {
        store.pushMessage({
          type: 'follow_up',
          documentId: document.id,
          questions: status.suggestion.follow_up_questions,
        });
      }
    }
  }, 2000);  // Poll every 2 seconds
}
```

```typescript
async function handleProactiveConfirm(msg: ProactiveMessage) {
  const store = useAIAdvisorStore.getState();
  store.updateMessageAction(msg.id, 'confirming');

  try {
    switch (msg.type) {
      case 'recurring_confirm':

    refreshStore.refreshAll();

    store.updateMessageAction(msg.id, 'error');

  }
}
```
